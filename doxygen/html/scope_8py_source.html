197 """
198 This class wrapps the xml node representing a FORTRAN scope

200 SCOPE_STMT = {'module': 'module-stmt',
201 'func': 'function-stmt',
202 'sub': 'subroutine-stmt',
203 'type': 'T-stmt',
204 'prog': 'program-stmt',
205 'interface': 'interface-stmt'}
206 SCOPE_CONSTRUCT = {'module': 'program-unit',
207 'func': 'program-unit',
208 'sub': 'program-unit',

213 def __init__(self, xml, scopePath='/', parentScope=None,
214 enableCache=False, tree=None, excludeContains=False):
215 """
216 :param xml: xml corresponding to this PYFTscope
217 :param scopePath: scope path ('/' separated string) of this node
218 :param parentScope: parent PYFTscope instance
219 :param enableCache: True to cache node parents
220 :param tree: an optional Tree instance
221 :param excludeContains: do not take into account the CONTAINS part
222 """
223 super().__init__(xml=xml, excludeContains=excludeContains)
224 self._mainScope = self if parentScope is None else parentScope._mainScope
225 self._path = scopePath
226 self._parentScope = parentScope
227 self.tree = Tree() if tree is None else tree
228 self._cacheParent = {}
229
230 if enableCache and parentScope is None:
231 # parent cache associated to the main scope
232 for node in self.iter():
233 for subNode in node:
234 self._cacheParent[id(subNode)] = node

280 def getParent(self, item, level=1):
281 """
282 :param item: item whose parent is to be searched
283 :param level: number of degrees (1 to get the parent, 2 to get
284 the parent of the parent...)
285 """
286 # pylint: disable=protected-access
287 def check(node):
288 # We check if the registered parent is still the right one
289 # node must be in its parent, and the parent chain must go to the root node
290 return node in self.mainScope._cacheParent.get(id(node), []) and \
291 (self.mainScope._cacheParent[id(node)] == self.mainScope._xml or
292 check(self.mainScope._cacheParent[id(node)]))
293
294 assert level >= 1
295 parent = None
296 if check(item):
297 parent = self.mainScope._cacheParent[id(item)]
298 else:
299 for node in self.mainScope.iter():
300 if item in list(node):
301 parent = node
302 if self.mainScope._cacheParent:
303 # _cacheParent not empty means that we want to use the caching system
304 self.mainScope._cacheParent[id(item)] = node
305 break
306 return parent if level == 1 else self.getParent(parent, level - 1)

308 def getSiblings(self, item, before=True, after=True):
309 """
310 :param item: item whose siblings are to be searched
311 :param before: returns siblings before
312 :param after: returns siblings after
313 By default before and after are True so that all siblings are returned
314 """
315
316 siblings = self.getParent(item).findall('./{*}*')
317 if not after:

391 def _getRecur(node, level, basePath=''):
392 # If node is the entire xml
393 if tag(node) == 'object':
394 usenode = node.find('./{*}file')
395 else:
396 usenode = node
397 results = []
398 for child in [child for child in usenode
399 if tag(child) in self.SCOPE_CONSTRUCT.values()]:
400 nodePath = self._getNodePath(child)
401 if excludeKinds is None or nodePath.split(':')[0] not in excludeKinds:
402 scopePath = self._getNodePath(child) if basePath in ('', '/') \
403 else basePath + '/' + nodePath
404 results.append(PYFTscope(child,
405 scopePath=scopePath, parentScope=self,
406 enableCache=False, tree=self.tree,
407 excludeContains=excludeContains))
408 if level != 1:
409 results.extend(_getRecur(child, level - 1, scopePath))
410 return results
411
412 if includeItself and tag(self) in self.SCOPE_CONSTRUCT.values():
413 itself = [self]
414 else:
415 itself = []
416
417 return _getRecur(self, level, self.pathpathpath) + itself

419 @debugDecor
420 def getScopeNode(self, scopePath, excludeContains=True, includeItself=True):
421 """
422 :param scopePath: scope path to search for
423 :param excludeContains: see getScopes
424 :param includeItself: include itself if self represent a "valid" scope (not a file)
425 :return: PYFTscope whose path is the path asked for

428 includeItself=includeItself)
429 if scope.path == scopePath]

445 @debugDecor
446 def getParentScopeNode(self, item, mustRaise=True):
447 """
448 :param item: item whose scope parent is to be searched
449 :param mustRaise: True to raise an exception if parent is not found
450 :return: the scope parent node of item
451 Example: if item is a call statement, result is the program-unit node
452 in which the call statement is
453 """
454 result = self.getParent(item)
455 while result is not None and not self.isScopeNode(result):
456 result = self.getParent(result)
457 if result is None and mustRaise:
458 raise PYFTError("The scope parent has not been found.")

461 @debugDecor
462 def getScopePath(self, item, includeItself=True):
463 """
464 :param item: item whose path must be determined
465 :param includeItself: include the item if it is a scope node
466 :return: the full path of the structure containing item
467 """
468 if includeItself and self.isScopeNode(item):
469 result = [self._getNodePath(item)]
