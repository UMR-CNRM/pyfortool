130  def convertTypesInCompute(self):
131  """
132  Convert STR%VAR into single local variable contained in compute (a-stmt)
133  and in if-then-stmt, else-if-stmt, where-stmt
134  e.g.
135  ZA = 1 + CST%XG ==> ZA = 1 + XCST_G
136  ZA = 1 + PARAM_ICE%XRTMIN(3) ==> ZA = 1 + XPARAM_ICE_XRTMIN3
137  ZRSMIN(1:KRR) = ICED%XRTMIN(1:KRR) => ZRSMIN(1:KRR) = ICEDXRTMIN1KRR(1:KRR)
138  IF(TURBN%CSUBG_MF_PDF=='NONE')THEN => IF(CTURBNSUBG_MF_PDF=='NONE')THEN
139 
140  RESTRICTION : works only if the r-component variable is contained in 1 parent structure.
141  Allowed for conversion : CST%XG
142  Not converted : TOTO%CST%XG (for now, recursion must be coded)
143  Not converted : TOTO%ARRAY(:) (shape of the array must be determined from E1)
144  """
145  def convertOneType(component, newVarList, scope):
146  # 1) Build the name of the new variable
147  objType = scope.getParent(component, 2) # The object STR%VAR
148  objTypeStr = alltext(objType).upper()
149  namedENn = objType.find('.//{*}N/{*}n')
150  structure = namedENn.text
151  variable = component.find('.//{*}ct').text.upper()
152  # If the variable is an array with index selection
153  # such as ICED%XRTMIN(1:KRR)
154  arrayIndices = ''
155  arrayRall = objType.findall('.//{*}array-R')
156  if len(arrayRall) > 0:
157  arrayR = copy.deepcopy(arrayRall[0]) # Save for the declaration
158  txt = alltext(arrayR).replace(',', '')
159  txt = txt.replace(':', '')
160  txt = txt.replace('(', '')
161  txt = txt.replace(')', '')
162  arrayIndices = arrayIndices + txt
163  elif len(objType.findall('.//{*}element-LT')) > 0:
164  # Case with single element such as ICED%XRTMIN(1)
165  for elem in objType.findall('.//{*}element'):
166  arrayIndices = arrayIndices + alltext(elem)
167  newName = variable[0] + structure + variable[1:] + arrayIndices
168  newName = newName.upper()
169 
170  # 2) Replace the namedE>N>n by the newName and delete R-LT
171  # except for array with index selection (R-LT is moved)
172  namedENn.text = newName
173  objType.remove(objType.find('.//{*}R-LT'))
174  if len(arrayRall) > 0:
175  objType.insert(1, arrayR)
176 
177  # 3) Add to the list of not already present for declaration
178  if newName not in newVarList:
179  if len(arrayRall) == 0:
180  newVarList[newName] = (None, objTypeStr)
181  else:
182  newVarList[newName] = (arrayR, objTypeStr)
183 
184  scopes = self.getScopes()
185  if scopes[0].path.split('/')[-1].split(':')[1][:4] == 'MODD':
186  return
187  for scope in [scope for scope in scopes
188  if 'sub:' in scope.path and 'interface' not in scope.path]:
189  newVarList = {}
190  for ifStmt in (scope.findall('.//{*}if-then-stmt') +
191  scope.findall('.//{*}else-if-stmt') +
192  scope.findall('.//{*}where-stmt')):
193  compo = ifStmt.findall('.//{*}component-R')
194  if len(compo) > 0:
195  for elcompo in compo:
196  convertOneType(elcompo, newVarList, scope)
197 
198  for aStmt in scope.findall('.//{*}a-stmt'):
199  # Exclude statements in which the component-R is in E1
200  # (e.g. PARAMI%XRTMIN(4) = 2)
201  if len(aStmt[0].findall('.//{*}component-R')) == 0: # E1 is the first son of aStmt
202  compoE2 = aStmt.findall('.//{*}component-R')
203  if len(compoE2) > 0:
204  # Exclude stmt from which E2 has only 1 named-E/{*}N/{*}n e.g. IKB = D%NKB
205  # warning, it does not handle yet op in simple statement
206  # such as ZEXPL = 1.- TURBN%XIMPL
207  # Include stmt from which E2 has 1 named-E/{*}N/{*}n AND E1 is an array;
208  # e.g. ZDELTVPT(JIJ,JK)=CSTURB%XLINF
209  nbNamedEinE2 = len(aStmt.findall('.//{*}E-2')[0].findall('.//{*}named-E/' +
210  '{*}N/{*}n'))
211  if nbNamedEinE2 > 1 or nbNamedEinE2 == 1 and \
212  len(aStmt[0].findall('.//{*}R-LT')) == 1:
213  for elcompoE2 in compoE2:
214  convertOneType(elcompoE2, newVarList, scope)
215 
216  # Add the declaration of the new variables and their affectation
217  for el, var in newVarList.items():
218  if el[0].upper() == 'X' or el[0].upper() == 'P' or el[0].upper() == 'Z':
219  varType = 'REAL'
220  elif el[0].upper() == 'L' or el[0].upper() == 'O':
221  varType = 'LOGICAL'
222  elif el[0].upper() == 'N' or el[0].upper() == 'I' or el[0].upper() == 'K':
223  varType = 'INTEGER'
224  elif el[0].upper() == 'C':
225  varType = 'CHARACTER(LEN=LEN(' + var[1] + '))'
226  else:
227  raise PYFTError('Case not implemented for the first letter of the newVarName' +
228  el + ' in convertTypesInCompute')
229  varArray = ''
230  # Handle the case the variable is an array
231  if var[0]:
232  varArray = ', DIMENSION('
233  for i, sub in enumerate(var[0].findall('.//{*}section-subscript')):
234  if len(sub.findall('.//{*}upper-bound')) > 0:
235  dimSize = simplifyExpr(
236  alltext(sub.findall('.//{*}upper-bound')[0]) +
237  '-' + alltext(sub.findall('.//{*}lower-bound')[0]) +
238  ' + 1')
239  elif len(sub.findall('.//{*}lover-bound')) > 0:
240  dimSize = simplifyExpr(alltext(sub.findall('.//{*}lower-bound')[0]))
241  else: # Case XRTMIN(:)
242  dimSize = 'SIZE(' + var[1] + ',' + str(i+1) + ')'
243  varArray = ', DIMENSION(' + dimSize + ','
244  varArray = varArray[:-1] + ')'
245  scope.addVar([[scope.path, el, varType + varArray + ' :: ' + el, None]])
246 
247  # Affectation
248  stmtAffect = createExpr(el + "=" + var[1])[0]
249  scope.insertStatement(scope.indent(stmtAffect), first=True)

315  def addMPPDB_CHECKS(self):
316 
317  """
318  Add MPPDB_CHEKS on all intent REAL arrays on subroutines.
319  ****** Not applied on modd_ routines. ********
320  Handle optional arguments.
321  Example, for a BL89 routine with 4 arguments, 1 INTENT(IN),
322  2 INTENT(INOUT), 1 INTENT(OUT), it produces :
323  IF (MPPDB_INITIALIZED) THEN
324  !Check all IN arrays
325  CALL MPPDB_CHECK(PZZ, "BL89 beg:PZZ")
326  !Check all INOUT arrays
327  CALL MPPDB_CHECK(PDZZ, "BL89 beg:PDZZ")
328  CALL MPPDB_CHECK(PTHVREF, "BL89 beg:PTHVREF")
329  END IF
330  ...
331  IF (MPPDB_INITIALIZED) THEN
332  !Check all INOUT arrays
333  CALL MPPDB_CHECK(PDZZ, "BL89 end:PDZZ")
334  CALL MPPDB_CHECK(PTHVREF, "BL89 end:PTHVREF")
335  !Check all OUT arrays
336  CALL MPPDB_CHECK(PLM, "BL89 end:PLM")
337  END IF
338  """
339  def addMPPDB_CHECK_statement(var, subRoutineName, strMSG='beg:'):
340  ifBeg, ifEnd, addD, addLastDim, addSecondDimType = '', '', '', '', ''
341  # Test if the variable is declared with the PHYEX D% structure,
342  # in that case, use the PHYEX MPPDB_CHECK interface
343  if var['as'][0][1]: # If not NoneType
344  if 'D%NIJT' in var['as'][0][1]:
345  addD = 'D,'
346  if len(var['as']) == 2:
347  # This handle 2D arrays with the last dim either D%NKT or anything else.
348  addLastDim = ', ' + var['as'][1][1]
349  if len(var['as']) >= 2:
350  # This adds information on the type of the second dimension :
351  # is it the vertical one or not, to remove extra points
352  if 'D%NK' in var['as'][1][1]:
353  addSecondDimType = ',' + '''"VERTICAL"'''
354  else:
355  addSecondDimType = ',' + '''"OTHER"'''
356  if 'MERGE' in var['as'][-1][1]: # e.g. MERGE(D%NKT,0,OCLOUDMODIFLM)
357  keyDimMerge = var['as'][-1][1].split(',')[2][:-1] # e.g. OCLOUDMODIFLM
358  ifBeg = 'IF (' + keyDimMerge + ') THEN\n'
359  ifEnd = '\nEND IF\n'
360  if var['opt']:
361  ifBeg = ifBeg + 'IF (PRESENT(' + var['n'] + ')) THEN\n IF (SIZE(' + \
362  var['n'] + ',1) > 0) THEN\n'
363  ifEnd = ifEnd + '\nEND IF\nEND IF'
364  argsMPPDB = var['n'] + ", " + "\"" + subRoutineName + " " + strMSG+var['n'] + "\""
365  return createExpr(ifBeg + "CALL MPPDB_CHECK(" + addD + argsMPPDB +
366  addLastDim + addSecondDimType + ")" + ifEnd)[0]
367  scopes = self.getScopes()
368  if scopes[0].path.split('/')[-1].split(':')[1][:4] == 'MODD':
369  return
370  for scope in scopes:
371  # Do not add MPPDB_CHEKS to :
372  # - MODULE or FUNCTION object,
373  # - interface subroutine from a MODI
374  # but only to SUBROUTINES
375  if 'sub:' in scope.path and 'func' not in scope.path and 'interface' not in scope.path:
376  subRoutineName = scope.path.split('/')[-1].split(':')[1]
377 
378  # Look for all intent arrays only
379  arraysIn, arraysInOut, arraysOut = [], [], []
380  for var in scope.varList:
381  if var['arg'] and var['as'] and 'TYPE' not in var['t'] and \
382  'REAL' in var['t'] and var['scopePath'] == scope.path:
383  if var['i'] == 'IN':
384  arraysIn.append(var)
385  if var['i'] == 'INOUT':
386  arraysInOut.append(var)
387  if var['i'] == 'OUT':
388  arraysOut.append(var)
389  # Check if there is any intent variables
390  if len(arraysIn) + len(arraysInOut) + len(arraysOut) == 0:
391  break
392 
393  # Add necessary module
394  scope.addModuleVar([(scope.path, 'MODE_MPPDB', None)])
395 
396  # Prepare some FORTRAN comments
397  commentIN = createElem('C', text='!Check all IN arrays', tail='\n')
398  commentINOUT = createElem('C', text='!Check all INOUT arrays', tail='\n')
399  commentOUT = createElem('C', text='!Check all OUT arrays', tail='\n')
400 
401  # 1) variables IN and INOUT block (beggining of the routine)
402  if len(arraysIn) + len(arraysInOut) > 0:
403  ifMPPDBinit = createExpr("IF (MPPDB_INITIALIZED) THEN\n END IF")[0]
404  ifMPPDB = ifMPPDBinit.find('.//{*}if-block')
405 
406  # Variables IN
407  if len(arraysIn) > 0:
408  ifMPPDB.insert(1, commentIN)
409  for i, var in enumerate(arraysIn):
410  ifMPPDB.insert(2 + i, addMPPDB_CHECK_statement(var, subRoutineName,
411  strMSG='beg:'))
412 
413  # Variables INOUT
414  if len(arraysInOut) > 0:
415  shiftLineNumber = 2 if len(arraysIn) > 0 else 1
416  ifMPPDB.insert(len(arraysIn) + shiftLineNumber, commentINOUT)
417  for i, var in enumerate(arraysInOut):
418  ifMPPDB.insert(len(arraysIn) + shiftLineNumber + 1 + i,
419  addMPPDB_CHECK_statement(var, subRoutineName,
420  strMSG='beg:'))
421 
422  # Add the new IN and INOUT block
423  scope.insertStatement(scope.indent(ifMPPDBinit), first=True)
424 
425  # 2) variables INOUT and OUT block (end of the routine)
426  if len(arraysInOut) + len(arraysOut) > 0:
427  ifMPPDBend = createExpr("IF (MPPDB_INITIALIZED) THEN\n END IF")[0]
428  ifMPPDB = ifMPPDBend.find('.//{*}if-block')
429 
430  # Variables INOUT
431  if len(arraysInOut) > 0:
432  ifMPPDB.insert(1, commentINOUT)
433  for i, var in enumerate(arraysInOut):
434  ifMPPDB.insert(2 + i, addMPPDB_CHECK_statement(var, subRoutineName,
435  strMSG='end:'))
436 
437  # Variables OUT
438  if len(arraysOut) > 0:
439  shiftLineNumber = 2 if len(arraysInOut) > 0 else 1
440  ifMPPDB.insert(len(arraysInOut) + shiftLineNumber, commentOUT)
441  for i, var in enumerate(arraysOut):
442  ifMPPDB.insert(len(arraysInOut) + shiftLineNumber + 1 + i,
443  addMPPDB_CHECK_statement(var, subRoutineName,
444  strMSG='end:'))
445 
446  # Add the new INOUT and OUT block
447  scope.insertStatement(scope.indent(ifMPPDBend), first=False)

451  """
452  Add specific allocations of local arrays on the fly for GPU
453  :param model : 'MESONH' or 'AROME' for specific objects related to the allocator or stack
454  :param stopScopes: scope paths where we stop to add stack
455  :param parser, parserOptions, wrapH: see the PYFT class
456 
457  Stacks are added to all routines called by the scopes listed in stopScopes
458  """
459  if model == 'AROME':
460  # The AROME transformation needs an additional parameter
461  # We apply the transformation only if the routine is called
462  # from a scope within stopScopes
463  for scope in [scope for scope in self.getScopes()
464  if scope.path in stopScopes or
465  self.tree.isUnderStopScopes(scope.path, stopScopes)]:
466  # Intermediate transformation, needs cpp to be completed
467  # This version would be OK if we didn't need to read again the files with fxtran
468  # after transformation
469  # nb = scope.modifyAutomaticArrays(
470  # declTemplate="temp({type}, {name}, ({shape}))",
471  # startTemplate="alloc({name})")
472 
473  # Full transformation, using CRAY pointers

475  # we do not call SOF with __FILE__ and __LINE__ because it breaks
476  # future reading with fxtran
477  nb = scope.modifyAutomaticArrays(
478  declTemplate="{type}, DIMENSION({shape}) :: {name}; " +
479  "POINTER(IP_{name}_, {name})",
480  startTemplate="IP_{name}_=YLSTACK%L(KIND({name})/4);" +
481  "YLSTACK%L(KIND({name})/4)=" +
482  "YLSTACK%L(KIND({name})/4)+" +
483  "KIND({name})*SIZE({name});" +
484  "IF(YLSTACK%L(KIND({name})/4)>" +
485  "YLSTACK%U(KIND({name})/4))" +
486  "CALL SOF('" + scope.getFileName() + ":{name}', " +
487  "KIND({name}))")

889  # the siblings of the parent of <f:o>*</f:o>

905  # <f:N>

908  # <f:R-LT>

910  # <f:element-LT>
911  # <f:element>
912  # ... ==> leftOfPow
913  # </f:element>,
914  # </f:element-LT>
915  # </f:parens-R>)
916  # </f:R-LT>
917  # </f:named-E>
918  nodeBRP = createExprPart('BR_P' + str(powerNumber))
919  element = createElem('element')
920  element.append(leftOfPow)
921  elementLT.append(element)

923  # except 2, 3 or 4 (powerBRList)
924  if tag(rightOfPow) != 'literal-E' or \
925  (tag(rightOfPow) == 'literal-E' and
926  int(alltext(rightOfPow).replace('.', '')) not in powerBRList):

929  # <f:n>BR_POW</f:n> or <f:n>BR_Pn</f:n>
930  # </f:N>

932  # <f:parens-R>(
933  # <f:element-LT>
934  # <f:element>
935  # ... ==> leftOfPow
936  # </f:element>,
937  # <f:element>
938  # ... ==> rightOfPow
939  # </f:element>
940  # </f:element-LT>
941  # </f:parens-R>)
942  # </f:R-LT>
943  # </f:named-E>
944  nodeBRP = createExprPart('BR_POW')

946  leftElement.append(leftOfPow)
947  rightElement = createElem('element')
