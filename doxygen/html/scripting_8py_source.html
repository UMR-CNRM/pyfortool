21 def task(filename):
22  """
23  Function to use on each file
24  :param clsPYFT: PYFT class to use
25  :param filename: file name
26  """
27  global PYFT
28  global allFileArgs
29  allArgs, orderedOptions = allFileArgs[filename]
30  try:
31  # Opening and reading of the FORTRAN file
32  pft = PYFT(filename, filename, parser=allArgs.parser,
33  parserOptions=getParserOptions(allArgs), verbosity=allArgs.logLevel,
34  wrapH=allArgs.wrapH,
35  enableCache=allArgs.enableCache)
36 
37  # apply the transformation in the order they were specified
38  for arg in orderedOptions:
39  logging.debug('Applying %s on %s', arg, filename)
40  applyTransfo(pft, arg, allArgs,
41  filename if filename == allArgs.plotCentralFile else None)
42  logging.debug(' -> Done')
43 
44  # Writing
45  if not allArgs.dryRun:
46  pft.write()
47 
48  # Closing and reporting
49  pft.close()
50  return (0, filename)
51 
52  except Exception as exc: # pylint: disable=broad-except
53  logging.error("The following error has occurred in the file %s", filename)
54  PYFT.unlockFile(filename, silence=True)
55  traceback.print_exception(exc, file=sys.stdout)
56  sys.stdout.flush()
57  return (1, filename)

168 def getArgs(parser):
169  """
170  Parse arguments and interpret the --optsByEnv option
171  :param parser: argparse parser
172  :return: a tuple with
173  - an argparse namespace containing common arguments (not using the --optsEnv option)
174  - a function taking a filename as input and returning
175  - an argparse namespace with the common arguments and the ones added by
176  interpreting the --optsEnv option
177  - an ordered list of arguments
178  """
179  args = parser.parse_args()
180 
181  def getFileArgs(filename=args.INPUT if hasattr(args, 'INPUT') else None):
182  """
183  :param filename: name of source code file
184  :return: argparse namespace to use with this file and
185  a list given the order in which the arguments were provided
186  """
187  # Decode the --optsByEnv option
188  arguments = sys.argv[1:]
189  if args.optsByEnv is not None:
190  extra = ''
191  for line in [] if args.optsByEnv is None else os.environ[args.optsByEnv].split('\n'):
192  if ':=:' in line:
193  if re.match(line.split(':=:')[0], filename):
194  extra = line.split(':=:')[1]
195  else:
196  extra = line
197  arguments.extend(shlex.split(extra))
198 
199  # Compute the ordered list
200  updateCnt = False
201  optList = []
202  for arg in arguments:
203  if arg.startswith('--') and arg not in optList:
204  if arg in ARG_UPDATE_CNT:
205  if not updateCnt:
206  updateCnt = True
207  optList.append(arg)
208  else:
209  optList.append(arg)
210 
211  return parser.parse_args(arguments), optList
212 
213  return args, getFileArgs
