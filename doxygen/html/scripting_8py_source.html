21def getArgs(parser):
22 """
23 Parse arguments and interpret the --optsByEnv option
24 :param parser: argparse parser
25 :return: a tuple with
26 - an argparse namespace containing common arguments (not using the --optsEnv option)
27 - a function taking a filename as input and returning
28 - an argparse namespace with the common arguments and the ones added by
29 interpreting the --optsEnv option
30 - an ordered list of arguments
31 """
32 args = parser.parse_args()
33
34 def getFileArgs(filename=args.INPUT if hasattr(args, 'INPUT') else None):
35 """
36 :param filename: name of source code file
37 :return: argparse namespace to use with this file and
38 a list given the order in which the arguments were provided
39 """
40 # Decode the --optsByEnv option
41 arguments = sys.argv[1:]
42 if args.optsByEnv is not None:
43 extra = ''
44 for line in [] if args.optsByEnv is None else os.environ[args.optsByEnv].split('\n'):
45 if ':=:' in line:
46 if re.match(line.split(':=:')[0], filename):
47 extra = line.split(':=:')[1]
48 else:
49 extra = line
50 arguments.extend(shlex.split(extra))
51
52 # Compute the ordered list
53 updateCnt = False
54 optList = []
55 for arg in arguments:
56 if arg.startswith('--') and arg not in optList:
57 if arg in ARG_UPDATE_CNT:
58 if not updateCnt:
59 updateCnt = True
60 optList.append(arg)
61 else:
62 optList.append(arg)
63
64 return parser.parse_args(arguments), optList
65
66 return args, getFileArgs

69def getParserOptions(args):
70 """
71 Get the options to use for the fxtran parser
72 :param args: arguments parsed by the argparse parser
73 """
74 if args.parserOption is None:
75 parserOptions = PYFT.DEFAULT_FXTRAN_OPTIONS.copy()
76 else:
77 parserOptions = [el for elements in args.parserOption for el in elements]
78 if args.addIncludes:
79 parserOptions = [opt for opt in parserOptions if opt not in ('-no-include', '-noinclude')]
80 return parserOptions

83def getDescTree(args, cls=Tree):
84 """
85 get the Tree object built with the parsed arguments
86 :param args: arguments parsed by the argparse parser
87 :param cls: class to use (usefull for manager)
88 :return: a Tree instance
89 """
90 parserOptions = getParserOptions(args)

100def updateParser(parser, withInput, withOutput, withXml, withPlotCentralFile, treeIsOptional,
101 nbPar):
102 """
103 Updates an argparse parser with arguments common to all the different tools
104 :param parser: parser in which arguments are added
105 :param withOutput: do we need the INPUT argument
106 :param withOutput: do we need the OUTPUT argument
107 :param withXml: do we need to be able to define an XML output file
108 :param withPlotCentralFile: to add the --plotCentralFile argument
109 :param treeIsOptional: is the --tree argument optional?
110 :param nbPar: number of parallel processes
111 """
112
113 # ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! !
114 # IMPORTANT NOTE
115 # Argument order matters but argparse is not able to give the order
116 # Therefore, arguments are processed twice. The first time by argparse to fully decode them.
117 # The a second pass is made direcly on sys.argv. This mechanism has two implications:
118 # allow_abbrev must be set to False in ArgumentParser
119 # only long argument options are allowed (begining with two dashes)
120 # ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! ! !
121 assert not parser.allow_abbrev, 'parser must be created with allow_abbrev=False'
122
123 parser.add_argument('--version', action='version',
124 version='%(prog)s {version}'.format(version=__version__))
125 parser.add_argument('--simplify', default=False, action='store_true',
126 help='After a deletion, recursively deletes the code ' +
127 'and variables that have become useless')
128 parser.add_argument('--logLevel', default='warning',
129 help='Provide logging level. Example --logLevel debug (default is warning)')
130 parser.add_argument('--enableCache', default=False, action='store_true',
131 help='Precompute parent of each xml node and store the result')
132 if nbPar:
133 parser.add_argument('--nbPar', default=cpu_count(), type=int,
134 help='Number of parallel processes, 0 to get as many processes ' +
135 'as the number of cores (default=0)')
136 parser.add_argument('--optsByEnv', default=None, type=str,
137 help='Name of the environment variable containing additional arguments ' +
138 'to use. These arguments are processed after all other arguments. ' +
139 'The variable can contain a multi-lines string. The ' +
140 'variable is read line by line and the last applicable line is ' +
141 'used. A line can take one of these two forms: ' +
142 '1) "FILE_DESCRIPTOR:=:OPTIONS" (where FILE_DESCRIPTOR is a ' +
143 'regular expression to test against the filename. If there ' +
144 'is a match, the OPTIONS can be used for the file) and ' +
145 '2) "OPTIONS" (if the line doesn\'t contain the FILE_DESCRIPTOR ' +
146 'part, it applies to all source code).')
147
148 # Inputs and outputs
149 updateParserInputsOutputs(parser, withInput, withOutput, withXml)
150
151 # fxtran
152 updateParserFxtran(parser)
153
154 # Variables
155 updateParserVariables(parser)
156
157 # Cosmetics
158 updateParserCosmetics(parser)
159
160 # Applications
161 updateParserApplications(parser)
162
163 # openACC
164 updateParserOpenACC(parser)
165
166 # Checks
167 updateParserChecks(parser)
168
169 # Statements
170 updateParserStatements(parser)
171
172 # Misc
173 updateParserMisc(parser)
174
175 # Tree
176 updateParserTree(parser, withPlotCentralFile, treeIsOptional)
177
178 # Preprocessot
179 updateParserPreprocessor(parser)

562def applyTransfoVariables(pft, arg, args, simplify, parserOptions, stopScopes):
563 """
564 Apply variables transformations on a PYFT instance
565 :param pft: PYFT instance
566 :param arg: argument to deal with
567 :param args: parsed argparsed arguments
568 :param simplify: kwargs to simplify
569 :param parserOptions: fxtran parser options
570 :param stopScopes: upper limit in call tree for some transformations
571 """
572 if arg == '--showVariables':
573 pft.varList.showVarList()
574 elif arg == '--attachArraySpecToEntity':
575 pft.attachArraySpecToEntity()
576 elif arg == '--removeVariable':
577 pft.removeVar(args.removeVariable, **simplify)
578 elif arg == '--addVariable':
